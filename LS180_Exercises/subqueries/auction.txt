#1 ---------------------------------------------------
Create a new database called auction, with 3 tables: bidders, items, and bids
- bidders: id serial (PK), name text (NOT NULL)

- items: id serial (PK), name text (NOT NULL), 
         initial_price numeric (0.00 - 1000.00, NOT NULL),
         sales_price numeric (0.00 - 1000.00)

- bids: id serial (PK), 
        bidder_id + item_id (FK, integer, NOT NULL, CASCADE ON DELETE)
        - index (bidder_id + item_id)
        amount (numeric, 0.00-1000.00, NOT NULL) 

Then use \copy to import data from the given files into the tables.

  createdb auction

  CREATE TABLE bidders (
    id serial PRIMARY KEY,
    name text NOT NULL
  );

  CREATE TABLE items (
    id serial PRIMARY KEY,
    name text NOT NULL,
    initial_price numeric(6, 2) NOT NULL CHECK (initial_price BETWEEN 0.00 AND 1000.00),
    sales_price numeric(6, 2) CHECK (sales_price BETWEEN 0.00 AND 1000.00)
  );

  CREATE TABLE bids (
    id serial PRIMARY KEY,
    bidder_id integer NOT NULL REFERENCES bidders ON DELETE CASCADE,
    item_id integer NOT NULL REFERENCES items ON DELETE CASCADE,
    amount numeric (6, 2) NOT NULL CHECK (amount BETWEEN 0.00 AND 1000.00)
  );

  CREATE INDEX bids_index ON bids (bidder_id, item_id);

  \copy bidders FROM 'bidders.csv' WITH CSV HEADER
  \copy items FROM 'items.csv' WITH CSV HEADER
  \copy bids FROM 'bids.csv' WITH CSV HEADER
